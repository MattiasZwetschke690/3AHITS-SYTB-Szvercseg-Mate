## Directory Struktur

Shell script:

```bash
#!/bin/bash
 
# Argument speichern
name=$1
 
# Hauptordner erstellen
mkdir -p "$name"
 
# Schleife für 2 Unterordner
for i in 01 02
do
    dir="$name/${name}_$i"
    mkdir -p "$dir"
 
    # Schleife für 3 Dateien
    for j in 1 2 3
    do
        touch "$dir/${name}.${i}.${j}.txt"
    done
done
```

## Skript Generator
shell script: 
```bash 
#!/bin/bash

name=$1
file="${name}.sh"

# Datei erstellen und Inhalt schreiben
echo "#!/bin/env sh" > "$file"
echo "" >> "$file"
echo "echo \"$name Skript\"" >> "$file"
echo "" >> "$file"
echo "# write your script here" >> "$file"

# ausführbar machen
chmod +x "$file"
```

 ## Headline Cat
 shell script:
 ```bash
 #!/bin/bash

# result.txt leeren/neu erstellen
> result.txt

for file in "$@"
do
    echo "== $file ==========================================" >> result.txt
    cat "$file" >> result.txt
    echo "" >> result.txt
done
 ```
